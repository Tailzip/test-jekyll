Getting Started

{% code bash %}# Susy only needs SASS, but was built with Compass so install it! gem install compass # You might need the alpha version of Compass, if you get errors when using susy run : gem install compass --pre # Install susy gem install susy # If you want to use the mixin susy-breakpoint(), you also need to install Breakpoint gem install breakpoint{% endcode %}
{% code ruby %}# Update config.rb require 'susy' # Require breakpoint if needed require 'breakpoint'{% endcode %} {% code javascript %}// Or update your gulpfile.js .pipe(compass({ require : ['susy', 'breakpoint'] }));{% endcode %}
{% code scss %}// Import breakpoint (if you need it) @import 'breakpoint'; // Import susy @import 'susy';{% endcode %}

First, Last & Full

{% code scss %}// FIRST #susy-container { /* * /!\ Override default settings in the entire .scss file */ @include layout(before); //change gutter position to before @include nested(12) { header { @include span(first 3); } nav { @include span(9); } article { @include span(3 break); } footer { @include span(9); } } } // LAST #susy-container { /* * /!\ Override default settings in the entire .scss file */ @include layout(after); //change gutter position to after @include nested(12) { header { @include span(9); } nav { @include span(last 3); } article { @include span(9 break); } footer { @include span(3); } } } // FULL #susy-container { @include nested(12) { header { @include span(full); } } } // COOL CLASSES .first { @include first; } .last { @include last; } .full { @include full; }{% endcode %}
